131 private void Vertex_MouseMove(object sender, MouseEventArgs e)
132 {

134 {
135 // Отримання нових координат вершини
136 double newX = e.GetPosition(graphCanvas).X - (selectedVertex.ActualWidth / 2);

147
148 if (newY < 0)
149 newY = 0;
150 else if (newY + selectedVertex.ActualHeight > canvasHeight)
151 newY = canvasHeight - selectedVertex.ActualHeight;
152
153 label.Content = $"Selected Vertex: {selectedVertex.Content}\nPosition (X:{newX} Y:{newY})";
154 // Зміна позиції вершини
155 Canvas.SetLeft(selectedVertex, newX);

294 for (int i = removedIndex; i < vertices.Count; i++)
295 {
296 vertices[i].VertexNumber -= 1;
297 vertices[i].Content = vertices[i].VertexNumber;
298 }
299 }

590 public override void UpdatePosition()
591 {
592 double startX = StartVertex.CenterX;
593 double startY = StartVertex.CenterY;
594 double endX = EndVertex.CenterX;
595 double endY = EndVertex.CenterY;
596
597 double angle = Math.Atan2(endY - startY, endX - startX);
598
599 double offset = StartVertex.ActualHeight / 2;
600
601 double lineStartX = startX + Math.Cos(angle) * offset;
602 double lineStartY = startY + Math.Sin(angle) * offset;
603 double lineEndX = endX - Math.Cos(angle) * offset;
604 double lineEndY = endY - Math.Sin(angle) * offset;
605
606 line.X1 = lineStartX;
607 line.Y1 = lineStartY;
608 line.X2 = lineEndX;
609 line.Y2 = lineEndY;
610
611 double arrowheadSize = 10;
612 double halfArrowheadSize = arrowheadSize / 2;
613 double arrowheadOffset = StartVertex.ActualHeight / 2 + halfArrowheadSize;
614
615 double arrowheadX = lineStartX + (lineEndX - lineStartX) / 1.02;
616 double arrowheadY = lineStartY + (lineEndY - lineStartY) / 1.02;
617
618 arrowhead.Width = arrowheadSize;
619 arrowhead.Height = arrowheadSize;
620 Canvas.SetLeft(arrowhead, arrowheadX - halfArrowheadSize);
621 Canvas.SetTop(arrowhead, arrowheadY - halfArrowheadSize);
622 arrowhead.RenderTransform = new RotateTransform(angle * (180 / Math.PI), halfArrowheadSize, halfArrowheadSize);
623
624 Canvas.SetZIndex(this, Math.Min(Canvas.GetZIndex(StartVertex), Canvas.GetZIndex(EndVertex)) - 1);
625 }
